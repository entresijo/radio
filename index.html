const STREAM_URL = "https://uk15freenew.listen2myradio.com/live.mp3?typeportmount=ice_4243_stream_737103432" const META_URL = "https://spring-sun-cde9.entresijoradio.workers.dev/" const btn = document.getElementById("btn") const player = document.getElementById("player") const onair = document.getElementById("onair") const track = document.getElementById("track") const listeners = document.getElementById("listeners") let playing = false function setOnAir(state) { if (state) { onair.className = "onair on" onair.innerText = "● EN EL AIRE" } else { onair.className = "onair off" onair.innerText = "● FUERA DEL AIRE" } } async function load() { try { const res = await fetch(META_URL, { cache: "no-store" }) const data = await res.json() if (data.error || !data.song) { setOnAir(false) track.innerText = "sin señal" listeners.innerText = "" return } setOnAir(true) track.innerText = data.song listeners.innerText = data.listeners ? "👥 " + data.listeners : "" } catch(e) { console.error("error metadata:", e) setOnAir(false) track.innerText = "error: " + e.message } } btn.onclick = () => { if (!playing) { player.src = STREAM_URL player.play().catch(err => { track.innerText = "error stream: " + err.message }) btn.innerText = "⏸" playing = true } else { player.pause() btn.innerText = "▶" playing = false } } player.addEventListener("playing", () => { if (playing) setOnAir(true) }) player.addEventListener("error", () => { track.innerText = "stream no disponible" }) setInterval(load, 15000) load()
